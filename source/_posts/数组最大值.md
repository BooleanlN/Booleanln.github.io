### 数组最大值

求数组最大值的常用方法总结

```js
// 排序方法
let arr = [1,34,21,41,22,23,7,3,9,10,1,1,1,1]
let compare = function (a,b) {
	if (a>b)return -1
	else if (a<b) return 1
	else return 0
}
arr.sort(compare)
console.log(arr[0])
// Math.max
arr = [1,34,21,41,22,23,7,3,9,10,1,1,1,1]
let max = Math.max.apply(null,arr)
console.log(max)
//reduce
arr = [1,34,21,41,22,23,7,3,9,10,1,1,1,1]
max = arr.reduce((currentMax,item) => {
	return item>currentMax ? item : currentMax
})
console.log(max)
// ES6
arr = [1,34,21,41,22,23,7,3,9,10,1,1,1,1]
max = Math.max(...arr)
console.log(max)
```
